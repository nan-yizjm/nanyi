# 122.买卖股票最佳时机II
```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        if(prices.size()<=1) return 0;
        int profit=prices[1]-prices[0];
        int r=-1,sum=0;
        if(profit>0)
        {
            r=prices[0];
            if(prices.size()==2)
                return profit;
        }
        for(int i=1;i<prices.size()-1;i++)
        {
            profit=prices[i+1]-prices[i];
            if(profit<0&&r!=-1)
            {
                sum=sum+prices[i]-r;
                r=-1;
            }
            else if(r==-1&&profit>0)
                r=prices[i];
            if(i==prices.size()-2&&r!=-1)
                sum=sum+prices[i+1]-r;
        }
        return sum;
    }
};
```
```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int result=0;
        for(int i=1;i<prices.size();i++)
            result+=max(prices[i]-prices[i-1],0);
        return result;
    }
};
```

这道题的贪心解法非常简单，思路巧妙，代码简洁，将问题转化为求和每一天的正利润即可，通过这个局部最优来达到全局最优。
